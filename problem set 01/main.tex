\begin{frame}{What Is Market Design?}
\begin{itemize}
    \item Markets are defined by rules, not just supply and demand
    \item Poor rules can cause inefficiency even without monopoly power
    \item Market design studies how to diagnose and fix such failures
\end{itemize}

% Slide 3
\begin{frame}{Three Goals of Market Design}
Market design research focuses on:
\begin{enumerate}
    \item Diagnosing market failures
    \item Evaluating and comparing designs
    \item Proposing improved mechanisms
\end{enumerate}
\end{frame}

% Slide 4
\begin{frame}{Diagnosing Market Failures}
Common failures caused by bad design:
\begin{itemize}
    \item Lack of coordination
    \item Strategic misreporting of preferences
    \item Participation failures
    \item Rent-seeking and institutional frictions
\end{itemize}
\end{frame}

% Slide 5
\begin{frame}{Example: Coordination Failures}
\begin{itemize}
    \item Decentralized markets may leave agents unmatched
    \item Centralized mechanisms can improve outcomes
    \item Example: school choice, kidney exchange
\end{itemize}
\end{frame}



% Slide 7
\begin{frame}{Evaluating Market Designs}
\small
Key questions:
\begin{itemize}
    \item How do different mechanisms affect welfare?
    \begin{itemize}
        \item Efficiency / total surplus, match quality, and distributional outcomes (who gains/loses)
    \end{itemize}

    \item What are the tradeoffs between efficiency and incentives?
    \begin{itemize}
        \item Strategy-proofness vs. efficiency: stronger incentives to tell the truth can change achievable outcomes
        \item Practical goal: good performance with realistic behavior and high participation
    \end{itemize}

    \item How sensitive are outcomes to strategic behavior?
    \begin{itemize}
        \item Robustness to gaming, information differences, and uneven sophistication across participants
    \end{itemize}
\end{itemize}
\end{frame}

% Slide 8
\begin{frame}{Role of Empirical Methods}
\begin{itemize}
    \item Structural models estimate preferences from data
    \item Enable counterfactual design comparisons
    \item Bridge theory and real-world performance
\end{itemize}
\end{frame}

% Slide 9
\begin{frame}{Designing Better Markets}
\begin{itemize}
    \item Market designers act as “economic engineers”
    \item Theory, data, and institutional detail are complementary
    \item Successful designs improve coordination and incentives
\end{itemize}
\end{frame}

% Slide 10
\begin{frame}{Adoption Challenges}
\begin{itemize}
    \item Incumbents may benefit from inefficient designs
    \item Political and regulatory constraints matter
    \item Good designs are not always voluntarily adopted
\end{itemize}
\end{frame}

% Slide 11
\begin{frame}{Key Takeaways}
\begin{itemize}
    \item Market failures often stem from bad rules
    \item Design and market power interact
    \item Empirical IO tools are central to modern market design
\end{itemize}
\end{frame}
